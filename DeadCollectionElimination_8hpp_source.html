208  // Transformation
209  bool transform(set<llvm::Value *> &&dead_values) {
210  // Drop all references to dead values that are Instructions.
211  set<llvm::Value *> values_ready_to_delete = {};
212  for (auto dead_value : dead_values) {
213  // Sanity check.
214  if (!dead_value) {
215  continue;
216  }
217 
218  // If this is an instruction, let's drop all references first.
219  if (auto dead_inst = dyn_cast<llvm::Instruction>(dead_value)) {
220  dead_inst->removeFromParent();
221  dead_inst->dropAllReferences();
222  }
223 
224  values_ready_to_delete.insert(dead_value);
225  }
226 
227  // Delete all values.
228  for (auto dead_value : values_ready_to_delete) {
229  dead_value->deleteValue();
230  }
231 
232  return true;
233  }
