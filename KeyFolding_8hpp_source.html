59  static inline bool value_escapes(llvm::Value &V) {
60  for (auto *user : V.users()) {
61  if (auto *user_as_call = dyn_cast_or_null<llvm::CallBase>(user)) {
62  // If the user is a memoir instruction, continue.
63  if (auto *user_as_memoir = MemOIRInst::get(*user_as_call)) {
64  continue;
65  }
66 
67  // Get the called function.
68  auto *called_function = user_as_call->getCalledFunction();
69 
70  // If it is an indirect call, then the value escapes.
71  // TODO: improve this by using NOELLE's complete call graph
72  if (called_function == nullptr) {
73  return true;
74  }
75 
76  // If it is empty, then the value escapes.
77  if (called_function->empty()) {
78  return true;
79  }
80  }
81  }
82 
83  return false;
84  }

100  static inline set<llvm::Value *> gather_possible_arguments(
101  llvm::Module &M,
102  llvm::Argument &A) {
103  set<llvm::Value *> possible_arguments = {};
104 
105  for (auto &F : M) {
106  for (auto &BB : F) {
107  for (auto &I : BB) {
108  if (auto *call = dyn_cast<llvm::CallBase>(&I)) {
109  // If this is a direct call, check if it calls the function we care
110  // about. If it does, stash the passed argument.
111  if (auto *called_function = call->getCalledFunction()) {
112  if (called_function == &F) {
113  auto *passed_argument =
114  (call->arg_begin() + A.getArgNo())->get();
115  possible_arguments.insert(passed_argument);
116  }
117  }
118  // Otherwise, this is an indirect call. Check if it _may_ call the
119  // function, if so, stash the passed argument.
120  else {
121  auto *function_type = call->getFunctionType();
122  if (function_type == F.getFunctionType()) {
123  auto *passed_argument =
124  (call->arg_begin() + A.getArgNo())->get();
125  possible_arguments.insert(passed_argument);
126  }
127  }
128  }
129  }
130  }
131  }
132 
133  return possible_arguments;
134  }

137  KeyFoldingInfo analyze(llvm::Module &M) {
138  KeyFoldingInfo info;
139  auto &key_folds = info.key_folds;
140  auto &assoc_accesses = info.assoc_accesses;
141  auto &key_to_index = info.key_to_index;
142 
143  // Find all associative and sequential collection allocations.
144  map<AssocArrayAllocInst *, AssocArrayType *> assoc_allocations = {};
145  map<SequenceAllocInst *, SequenceType *> seq_allocations = {};
146 
147  // For each function:
148  for (auto &F : M) {
149  // Ignore function declarations.
150  if (F.empty()) {
151  continue;
152  }
153 
154  // Check the function.
155  for (auto &BB : F) {
156  for (auto &I : BB) {
157  auto *memoir_inst = MemOIRInst::get(I);
158  if (memoir_inst == nullptr) {
159  continue;
160  }
161 
162  if (auto *seq_alloc = dyn_cast<SequenceAllocInst>(memoir_inst)) {
163  auto *seq_type =
164  dyn_cast<SequenceType>(&seq_alloc->getCollectionType());
165  MEMOIR_NULL_CHECK(seq_type,
166  "Sequence allocation is not of sequence type!");
167 
168  seq_allocations[seq_alloc] = seq_type;
169 
170  debugln("Found Sequence Alloc: ", *seq_alloc);
171 
172  } else if (auto *assoc_alloc =
173  dyn_cast<AssocArrayAllocInst>(memoir_inst)) {
174  auto *assoc_type =
175  dyn_cast<AssocArrayType>(&assoc_alloc->getCollectionType());
176  MEMOIR_NULL_CHECK(assoc_type,
177  "Assoc allocation is not of assoc type!");
178 
179  assoc_allocations[assoc_alloc] = assoc_type;
180 
181  debugln("Found Assoc Alloc: ", *assoc_alloc);
182  }
183  }
184  }
185  }
186 
187  debugln("Got assoc and seq allocations.");
188 
189  // Next, we need to check if there is any assoc-sequence pair for which all
190  // accesses to the assoc are via references to the elements of the sequence.
191 
192  // For each assoc allocation:
193  for (auto const &[assoc_alloc, assoc_type] : assoc_allocations) {
194  debugln("Visiting ", *assoc_alloc);
195 
196  // Check that the assoc type has a reference type for its key.
197  auto &key_type = assoc_type->getKeyType();
198  auto *key_ref_type = dyn_cast<ReferenceType>(&key_type);
199  if (key_ref_type == nullptr) {
200  debugln(" key is not a reference");
201  continue;
202  }
203  auto &referenced_type = key_ref_type->getReferencedType();
204  auto *referenced_struct_type = dyn_cast<StructType>(&referenced_type);
205  if (referenced_struct_type == nullptr) {
206  debugln(" key is not a struct reference");
207  continue;
208  }
209 
210  // Get the LLVM representation of the assoc alloc.
211  auto &llvm_assoc_alloc = assoc_alloc->getCallInst();
212 
213  // Inspect each access to this collection, recording the source of their
214  // key value for each.
215  auto &assoc_alloc_accesses = assoc_accesses[assoc_alloc];
216 
217  set<llvm::Value *> visited = {};
218  vector<llvm::Value *> worklist = {};
219  worklist.push_back(&llvm_assoc_alloc);
220  while (!worklist.empty()) {
221  // Pop an item of the worklist.
222  auto *workitem = worklist.back();
223  worklist.pop_back();
224 
225  // Check that this item hasn't been visited.
226  if (visited.find(workitem) != visited.end()) {
227  continue;
228  } else {
229  visited.insert(workitem);
230  }
231 
232  // Iterate over the uses of this item.
233  for (auto &use : workitem->uses()) {
234  auto *user = use.getUser();
235 
236  // Check that the user is an instruction. If not skip it.
237  auto *user_as_inst = dyn_cast<llvm::Instruction>(user);
238  if (user_as_inst == nullptr) {
239  continue;
240  }
241 
242  // Handle memoir instructions.
243  if (auto *user_as_memoir = MemOIRInst::get(*user_as_inst)) {
244  // If this is an assoc access instruction, fetch the key used to
245  // access and stash them.
246  if (auto *assoc_read = dyn_cast<AssocReadInst>(user_as_memoir)) {
247  auto &key_use = assoc_read->getKeyOperandAsUse();
248  assoc_alloc_accesses[assoc_read] = &key_use;
249  } else if (auto *assoc_write =
250  dyn_cast<AssocWriteInst>(user_as_memoir)) {
251  auto &key_use = assoc_write->getKeyOperandAsUse();
252  assoc_alloc_accesses[assoc_write] = &key_use;
253  } else if (auto *assoc_get =
254  dyn_cast<AssocGetInst>(user_as_memoir)) {
255  auto &key_use = assoc_get->getKeyOperandAsUse();
256  assoc_alloc_accesses[assoc_get] = &key_use;
257  }
258 
259  // If this is a Use/DefPHI, iterate on its resultant.
260  else if (auto *use_phi = dyn_cast<UsePHIInst>(user_as_memoir)) {
261  worklist.push_back(&use_phi->getCallInst());
262  } else if (auto *def_phi = dyn_cast<DefPHIInst>(user_as_memoir)) {
263  worklist.push_back(&def_phi->getCallInst());
264  }
265  }
266 
267  // Handle PHI Nodes, iterating on the resultant.
268  if (auto *use_as_phi = dyn_cast<llvm::PHINode>(user_as_inst)) {
269  worklist.push_back(use_as_phi);
270  }
271 
272  // Handle call instructions, iterating on the argument corresponding
273  // to this use.
274  if (auto *user_as_call = dyn_cast<llvm::CallBase>(user_as_inst)) {
275  auto argument_number = use.getOperandNo();
276 
277  // Fetch the called function, if this is a direct call, handle it.
278  auto *called_function = user_as_call->getCalledFunction();
279  if (called_function != nullptr) {
280  auto *argument = called_function->arg_begin() + argument_number;
281  worklist.push_back(argument);
282  continue;
283  }
284 
285  // If this is an indirect call, conservatively grab every function
286  // that _may_ be called.
287  auto *function_type = user_as_call->getFunctionType();
288  MEMOIR_NULL_CHECK(function_type,
289  "CallBase has NULL function type!");
290  for (auto &F : M) {
291  auto *other_function_type = F.getFunctionType();
292 
293  // If this function _could_ be called (becuase it shares a
294  // function type with the call), grab its argument and add it to
295  // the worklist.
296  if (other_function_type == function_type) {
297  auto *argument = F.arg_begin() + argument_number;
298  worklist.push_back(argument);
299  continue;
300  }
301  }
302  }
303  }
304  }

306  // Next, we will check each of the keys used to access this collection to
307  // see if it is _always_ derived from the same collection.
308  bool unknown_behavior = false;
309  set<llvm::Value *> collections_accessed = {};
310  for (auto const &[assoc_access, assoc_key] : assoc_alloc_accesses) {
311  // Check if the key is an instruction.
312  auto *key_as_inst = dyn_cast<llvm::Instruction>(assoc_key->get());
313  if (key_as_inst == nullptr) {
314  unknown_behavior = true;
315  break;
316  }
317 
318  // Check if the key is a memoir instruction.
319  if (auto *key_as_memoir = MemOIRInst::get(*key_as_inst)) {
320  // If the key is the result of an IndexGetInst, record the sequence
321  // accessed and the mapping from this key to the index accessed.
322  if (auto *key_as_index_get = dyn_cast<IndexGetInst>(key_as_memoir)) {
323  // Fetch information from the IndexGetInst.
324  auto &sequence_accessed = key_as_index_get->getObjectOperand();
325  auto &index_accessed = key_as_index_get->getIndexOfDimension(0);
326 
327  // Record the sequence variable that was accessed along with the
328  // key-to-index mapping.
329  collections_accessed.insert(&sequence_accessed);
330  key_to_index[assoc_key] = &index_accessed;
331 
332  // Keep on chugging.
333  continue;
334  }
335 
336  // If the key is the result of an AssocGetInst, record the sequence
337  // accessed and the mapping from this key to the key accessed.
338  if (auto *key_as_assoc_get = dyn_cast<AssocGetInst>(key_as_memoir)) {
339  // Fetch information from the IndexGetInst.
340  auto &assoc_accessed = key_as_assoc_get->getObjectOperand();
341  auto &key_accessed = key_as_assoc_get->getKeyOperand();
342 
343  // Record the sequence variable that was accessed along with the
344  // key-to-index mapping.
345  collections_accessed.insert(&assoc_accessed);
346  key_to_index[assoc_key] = &key_accessed;
347 
348  // Keep on chugging.
349  continue;
350  }
351  }
352 
353  // TODO: make this analysis more powerful by iterating along PHIs and
354  // building ValueExpressions for the key-to-index mapping.
355 
356  // Otherwise, record this as unknown behavior and leave.
357  unknown_behavior = true;
358  break;
359  }
360 
361  // If we found behavior that we couldn't analyze, then don't key fold
362  // here.
363  if (unknown_behavior) {
364  debugln(" found unknown behavior, skipping.");
365  continue;
366  }
367 
368  // Check that all collections accessed _must_ be the same allocation.
369  set<CollectionAllocInst *> collections_referenced = {};
370  visited.clear();
371  worklist.clear();
372  worklist.insert(worklist.end(),
373  collections_accessed.begin(),
374  collections_accessed.end());
375  while (!worklist.empty()) {
376  // Pop an item of the worklist.
377  auto *workitem = worklist.back();
378  debugln(*workitem);
379  worklist.pop_back();
380 
381  // Check that this item hasn't been visited.
382  if (visited.find(workitem) != visited.end()) {
383  continue;
384  } else {
385  visited.insert(workitem);
386  }
387 
388  // If the workitem is an instruction, handle each case, traversing UP
389  // the def-use chain.
390  if (auto *inst = dyn_cast<llvm::Instruction>(workitem)) {
391  if (auto *memoir_inst = MemOIRInst::get(*inst)) {
392  if (auto *seq_alloc = dyn_cast<SequenceAllocInst>(memoir_inst)) {
393  collections_referenced.insert(seq_alloc);
394  } else if (auto *assoc_alloc =
395  dyn_cast<AssocArrayAllocInst>(memoir_inst)) {
396  collections_referenced.insert(assoc_alloc);
397  } else if (auto *use_phi = dyn_cast<UsePHIInst>(memoir_inst)) {
398  worklist.push_back(&use_phi->getUsedCollection());
399  } else if (auto *def_phi = dyn_cast<DefPHIInst>(memoir_inst)) {
400  worklist.push_back(&def_phi->getDefinedCollection());
401  } else if (auto *seq_write_inst =
402  dyn_cast<IndexWriteInst>(memoir_inst)) {
403  worklist.push_back(&seq_write_inst->getObjectOperand());
404  } else if (auto *assoc_write_inst =
405  dyn_cast<AssocWriteInst>(memoir_inst)) {
406  worklist.push_back(&assoc_write_inst->getObjectOperand());
407  } else {
408  // TODO: add handling for slice and join instructions, this is
409  // tricky though and would be better served by insert/remove
410  // operators.
411  }
412  } else if (auto *phi = dyn_cast<llvm::PHINode>(inst)) {
413  for (auto &incoming_use : phi->incoming_values()) {
414  auto *incoming = incoming_use.get();
415  worklist.push_back(incoming);
416  }
417  } else if (auto *call = dyn_cast<llvm::CallBase>(inst)) {
418  // Round up all the possible callees and iterate on their returns.
419 
420  // Fetch the called function, if this is a direct call, handle it.
421  auto *called_function = call->getCalledFunction();
422  if (called_function != nullptr) {
423  auto returned_values = gather_returned_values(*called_function);
424  worklist.insert(worklist.end(),
425  returned_values.begin(),
426  returned_values.end());
427  continue;
428  }
429 
430  // If this is an indirect call, conservatively grab every function
431  // that _may_ be called.
432  auto *function_type = call->getFunctionType();
433  MEMOIR_NULL_CHECK(function_type,
434  "CallBase has NULL function type!");
435  for (auto &F : M) {
436  auto *other_function_type = F.getFunctionType();
437 
438  // If this function _could_ be called (becuase it shares a
439  // function type with the call), grab its argument and add it to
440  // the worklist.
441  if (other_function_type == function_type) {
442  auto returned_values = gather_returned_values(F);
443  worklist.insert(worklist.end(),
444  returned_values.begin(),
445  returned_values.end());
446  continue;
447  }
448  }
449  }
450  } else if (auto *arg = dyn_cast<llvm::Argument>(workitem)) {
451  // Round up all the possible callers and iterate on their arguments.
452  auto possible_arguments = gather_possible_arguments(M, *arg);
453  worklist.insert(worklist.end(),
454  possible_arguments.begin(),
455  possible_arguments.end());
456  } else {
457  collections_referenced.insert(nullptr);
458  }
459  }
460 
461  // If there is more than one sequence that may be referenced, skip this
462  // assoc.
463  if (collections_referenced.size() > 1) {
464  debugln(" keys of assoc may reference more than one sequence");
465  continue;
466  } else if (collections_referenced.size() == 0) {
467  debugln(" could not find any sequence allocations referenced.");
468  continue;
469  }
470 
471  // Otherwise, record the assoc-collection pair for transformation.
472  auto *collection_alloc = *(collections_referenced.begin());
473  if (collection_alloc == nullptr) {
474  debugln(" found an unhandled case");
475  continue;
476  }
477  key_folds[assoc_alloc] = collection_alloc;
478  }

480  debugln("Found the following key folding pairs.");
481

483  debugln(" ", *assoc_alloc);
484  debugln(" <==>");
485  debugln(" ", *collection_alloc);
486  debugln();

507  // Transformation driver.
508  bool transform(KeyFoldingInfo info) {
509  auto &key_folds = info.key_folds;
510  auto &assoc_accesses = info.assoc_accesses;
511  auto &key_to_index = info.key_to_index;
512 
513  // Track if we actually transformed the program.
514  bool transformed = false;
515 
516  // Stash for any instructions that will be removed by the transformation.
517  set<llvm::Instruction *> instructions_to_delete = {};
518 
519  // For key-fold pair.
520  for (auto const &[assoc_alloc, collection_alloc] : key_folds) {
521  if (auto *seq_alloc = dyn_cast<SequenceAllocInst>(collection_alloc)) {
522  // Convert the assoc allocation to be a sequence allocation the same
523  // size as the sequence allocation.
524  auto &seq_size = seq_alloc->getSizeOperand();
525 
526  // Check if the size is available at the assoc allocation site.
527  // NOTE: this is a little conservative to keep things simple: if the
528  // size is not available at the assoc allocation site, then we will
529  // _not_ perform key folding for this candidate.
530  llvm::Value *allocation_size;
531 
532  // If the size is a constant, it is available.
533  if (auto *size_as_const_int = dyn_cast<llvm::ConstantInt>(&seq_size)) {
534  allocation_size = size_as_const_int;
535  }
536  // If the size is an argument of the function containing the assoc
537  // allocation, it is available.
538  else if (auto *size_as_argument = dyn_cast<llvm::Argument>(&seq_size)) {
539  allocation_size = size_as_argument;
540  }
541  // If the size dominates the assoc allocation, it is available.
542  else if (false) {
543  debugln("Sequence size based on dominance is unimplemented");
544  allocation_size = nullptr;
545  }
546  // Otherwise, the size is not available, skip this pair.
547  else {
548  allocation_size = nullptr;
549  }
550 
551  // Check that we were able to find an allocation size.
552  if (allocation_size == nullptr) {
553  debugln("Sequence size is not available.");
554  continue;
555  }
556 
557  // Build a new sequence allocation.
558  MemOIRBuilder builder(*assoc_alloc);
559 
560  // The value operand _will_ be available, because we are inserting
561  // immediately before the existing allocation.
562  auto &elem_type = assoc_alloc->getValueOperand();
563 
564  auto *folded_sequence = builder.CreateSequenceAllocInst(&elem_type,
565  allocation_size,
566  "folded.");
567 
568  // Replace the assoc allocation with the folded one.
569  assoc_alloc->getCallInst().replaceAllUsesWith(
570  &folded_sequence->getCallInst());
571 
572  // Mark the old allocation for cleanup.
573  instructions_to_delete.insert(&assoc_alloc->getCallInst());
574 
575  // For each access, convert it to be an indexed access using the
576  // key-to-index map.
577  auto &accesses = assoc_accesses[assoc_alloc];
578  for (auto const &[access, key_use] : accesses) {
579  // Fetch the information about this access.
580  auto &access_call = access->getCallInst();
581  auto *access_function = access_call.getCalledFunction();
582  MEMOIR_NULL_CHECK(access_function, "Access function is NULL!");
583 
584  // Get the index variant of this function.
585  auto *replacement_function = get_index_function(*access_function);
586  MEMOIR_NULL_CHECK(replacement_function,
587  "Couldn't get the index function");
588 
589  // Replace the called function.
590  access_call.setCalledFunction(replacement_function);
591 
592  // Replace the key use with its corresponding index.
593  // TODO: enhance this to materialize a ValueExpression.
594  auto *folded_index = key_to_index[key_use];
595  key_use->set(folded_index);
596  }
597  } else if (auto *fold_assoc_alloc =
598  dyn_cast<AssocArrayAllocInst>(collection_alloc)) {
599  // Build a new assoc allocation.
600  MemOIRBuilder builder(*assoc_alloc, /*InsertAfter=*/false);
601 
602  // Get the key type that we are folding onto.
603  auto &key_type = fold_assoc_alloc->getKeyType();
604 
605  // The value type operand _will_ be available, because we are inserting
606  // immediately before the existing allocation.
607  auto &value_type = assoc_alloc->getValueOperand();
608 
609  auto *folded_alloc =
610  builder.CreateAssocArrayAllocInst(key_type, &value_type, "folded.");
611 
612  // Replace the assoc allocation with the folded one.
613  assoc_alloc->getCallInst().replaceAllUsesWith(
614  &folded_alloc->getCallInst());
615 
616  // Mark the old allocation for cleanup.
617  instructions_to_delete.insert(&assoc_alloc->getCallInst());
618 
619  // For each access, convert it to use the folded key.
620  auto &accesses = assoc_accesses[assoc_alloc];
621  for (auto const &[access, key_use] : accesses) {
622  // Replace the key use with its corresponding index.
623  // TODO: enhance this to materialize a ValueExpression.
624  auto *folded_index = key_to_index[key_use];
625  key_use->set(folded_index);
626  }
627  }
628  }
629 
630  // Cleanup instructions.
631  for (auto *inst : instructions_to_delete) {
632  inst->eraseFromParent();
633  }
634 
635  // Return, true if the program was transformed, false otherwise.
636  return transformed;
637  }
