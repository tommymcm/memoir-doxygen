52  // Given a start node, produce a vector that is the depth first
53  // traversal starting from that node.
54  vector<NodeType> depth_first_order(NodeType start) {
55  // procedure DFS_iterative(G, v) is
56  // let S be a stack
57  // S.push(v)
58  // while S is not empty do
59  // v = S.pop()
60  // if v is not labeled as discovered then
61  // label v as discovered
62  // for all edges from v to w in G.adjacentEdges(v) do
63  // S.push(w)
64 
65  vector<NodeType> out;
66  set<NodeType> discovered;
67  stack<NodeType> s;
68  s.push(start);
69  while (not s.empty()) {
70  auto v = s.top();
71  s.pop();
72 
73  // If v was not labeled as discovered
74  if (discovered.find(v) == discovered.end()) {
75  // label v as discovered
76  discovered.insert(v);
77  // add it to the list
78  out.push_back(v);
79 
80  for (auto &[w, _] : this->outgoing(v)) {
81  s.push(w);
82  }
83  }
84  }

111  bool topological_order_helper(NodeType node,
112  vector<NodeType> &order,
113  set<NodeType> &permanent_mark,
114  set<NodeType> &temporary_mark) {
115  if (permanent_mark.count(node) != 0) {
116  return true;
117  }
118 
119  if (temporary_mark.count(node) != 0) {
120  // We detected a cycle! Signal that the ordering has failed.
121  return false;
122  }
123 
124  // Temporarily mark the node.
125  temporary_mark.insert(node);
126 
127  // For each outgoing edge:
128  for (auto [out, _] : this->outgoing(node)) {
129  if (!this->topological_order_helper(out,
130  order,
131  permanent_mark,
132  temporary_mark)) {
133  return false;
134  }
135  }
136 
137  // Remove temporary mark from node.
138  temporary_mark.erase(node);
139 
140  // Permanently mark the node.
141  permanent_mark.insert(node);
142 
143  // Prepend the node to the order.
144  order.push_back(node);
145 
146  return true;
147  }

150  opt<vector<NodeType>> topological_order() {
151  vector<NodeType> order = {};
152  order.reserve(this->size());
153 
154  set<NodeType> permanent_mark = {};
155  set<NodeType> temporary_mark = {};
156 
157  // While there exists a node without a permanent mark:

159  auto node = *node_it;

181  void tarjan_scc_helper(NodeType n,
182  Components &components,
183  uint32_t &index,
184  map<NodeType, uint32_t> &indices,
185  map<NodeType, uint32_t> &lowlink,
186  vector<NodeType> &stack) {
187  indices[n] = index;
188  lowlink[n] = index;
189  ++index;
190  stack.push_back(n);
191 
192  // For each outgoing edge of n->m:
193  for (auto [to_node, _] : this->outgoing(n)) {
194  // If m has no index, recurse on it.
195  if (indices.count(to_node) == 0) {
196  tarjan_scc_helper(to_node, components, index, indices, lowlink, stack);
197  lowlink[n] = std::min(lowlink[n], lowlink[to_node]);
198  }
199 
200  // Otherwise, if m is on the stack, then n and m are in the same SCC,
201  // ignore this edge.
202  else if (std::find(stack.begin(), stack.end(), to_node) != stack.end()) {
203  lowlink[n] = std::min(lowlink[n], indices[to_node]);
204  }
205  }
206 
207  // If we have a root node, pop the stack and generate an SCC.
208  if (lowlink[n] == indices[n]) {
209  ConnectedComponent &scc = components[n];
210 
211  NodeType pop_node;
212  do {
213  pop_node = stack.back();
214  stack.pop_back();
215  scc.insert(pop_node);
216  } while (pop_node != n);
217  }
218 
219  return;
220  }
