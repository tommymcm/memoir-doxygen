35 struct TypeVariable : public Type {
36 public:
37  using TypeID = uint64_t;
38 
39  // Constructor
40  TypeVariable(TypeID id) : Type(TypeCode::OTHER), id(id) {}
41 
42  // Equality.
43  bool operator==(Type &T) const {
44  if (auto *tvar = dyn_cast<TypeVariable>(&T)) {
45  return tvar->id == this->id;
46  }
47  return false;
48  }
49 
50  // This class will only be used in the context of the base types and
51  // itself, so it is the only one that follows "other".
52  static bool classof(const Type *t) {
53  return (t->getCode() == TypeCode::OTHER);
54  }
55 
56  std::string toString(std::string indent = "") const override {
57  return "typevar(" + std::to_string(this->id) + ")";
58  }
59 
60 protected:
61  TypeID id;
