62  // Top-level driver.
63  bool run() {
64  // Fetch the live range analysis results.
65  const auto &live_ranges = LRAR.live_ranges();
66 
67  // For each live range result:
68  for (auto const &[collection, context_to_live_range] : live_ranges) {
69  for (auto const &[context, live_range] : context_to_live_range) {
70 
71  // If the context is defined, create a version of it unless it already
72  // exists.
73  if (context != nullptr) {
74  MEMOIR_UNREACHABLE(
75  "Context-sensitive dead-element elimination is unimplemented!");
76  }
77 
78  // If the live range is undefined, skip it.
79  if (live_range == nullptr) {
80  continue;
81  }
82 
83  // Get the definition instruction, if it exists.
84  auto *inst = dyn_cast<llvm::Instruction>(collection);
85  if (!inst) {
86  continue;
87  }
88 
89  debugln("DEE: ", *inst);
90  debugln(" ", *live_range);
91 
92  // Ignore upper/lower range if they are max/min.
93  auto &upper = live_range->get_upper();
94  bool ignore_upper = isa<EndExpression>(&upper);
95 
96  auto &lower = live_range->get_lower();
97  bool ignore_lower = false;
98  if (auto *lower_const = dyn_cast<ConstantExpression>(&lower)) {
99  if (auto *lower_const_int =
100  dyn_cast<llvm::ConstantInt>(&lower_const->getConstant())) {
101  if (lower_const_int->getZExtValue() == 0) {
102  ignore_lower = true;
103  }
104  }
105  }
106 
107  // If we are ignoring upper and lower, ignore the range.
108  if (ignore_upper && ignore_lower) {
109  debugln(" ignoring range");
110  continue;
111  }
112 
113  // If the upper and lower value expressions are available:
114  if ((ignore_upper || upper.isAvailable(*inst))
115  && (ignore_lower || lower.isAvailable(*inst))) {
116  // Materialize the upper and lower value expressions.
117  llvm::Value *materialized_upper = nullptr;
118  if (!ignore_upper) {
119  materialized_upper = upper.materialize(*inst);
120  if (!materialized_upper) {
121  continue;
122  }
123  }
124  llvm::Value *materialized_lower = nullptr;
125  if (!ignore_lower) {
126  materialized_lower = lower.materialize(*inst);
127  if (!materialized_lower) {
128  continue;
129  }
130  }
131 
132  // Perform the rewrite rule.
133  if (auto *write_inst = into<IndexWriteInst>(inst)) {
134 
135  // Fetch the index of the write instruction.
136  auto &index = write_inst->getIndexOfDimension(0);
137 
138  // We will first construct the conditional check on the index.
139  MemOIRBuilder builder(inst);
140  llvm::Value *cond = nullptr;
141  if (materialized_lower) {
142  auto *lower_cmp =
143  builder.CreateICmpULE(materialized_lower, &index);
144  cond = lower_cmp;
145  }
146  if (materialized_upper) {
147  auto *upper_cmp =
148  builder.CreateICmpULT(&index, materialized_upper);
149  if (cond) {
150  cond = builder.CreateAnd(upper_cmp, cond);
151  } else {
152  cond = upper_cmp;
153  }
154  }
155 
156  if (!cond) {
157  MEMOIR_UNREACHABLE("Could not create the comparison!");
158  }
159 
160  // Save the original basic block.
161  auto *original_bb = inst->getParent();
162 
163  // Then, construct a hammock to move the write to.
164  // NOTE: @new_terminator is the terminator of the "then" basic
165  // block.
166  auto *then_terminator =
167  llvm::SplitBlockAndInsertIfThen(/* Condition = */ cond,
168  /* Split Before = */ inst,
169  /* Unreachable = */ false);
170 
171  // Create a PHI at the split point.
172  builder.SetInsertPoint(inst);
173  auto *phi = builder.CreatePHI(inst->getType(), 2, "dee");
174 
175  // Move the instruction to the "then" basic block.
176  inst->moveBefore(then_terminator);
177 
178  // Replace uses of the write result with the PHI node.
179  inst->replaceAllUsesWith(phi);
180 
181  // Patch up the PHI for the "then" branch.
182  auto *then_bb = then_terminator->getParent();
183  phi->addIncoming(inst, then_bb);
184 
185  // Patch up the PHI for the "else" branch.
186  auto &original_collection = write_inst->getObjectOperand();
187  phi->addIncoming(&original_collection, original_bb);
188 
189  } else if (auto *insert_inst = into<SeqInsertInst>(inst)) {
190 
191  // Fetch the insertion point.
192  auto &index = insert_inst->getInsertionPoint();
193 
194  // We will first construct the conditional check on the index.
195  MemOIRBuilder builder(inst);
196  llvm::Value *cond = nullptr;
197  if (materialized_upper) {
198  auto *upper_cmp =
199  builder.CreateICmpULT(&index, materialized_upper);
200  cond = upper_cmp;
201  }
202 
203  if (!cond) {
204  MEMOIR_UNREACHABLE("Could not create the comparison!");
205  }
206 
207  // Save the original basic block.
208  auto *original_bb = inst->getParent();
209 
210  // Then, construct a hammock to move the insert to.
211  // NOTE: @new_terminator is the terminator of the "then" basic
212  // block.
213  auto *then_terminator =
214  llvm::SplitBlockAndInsertIfThen(/* Condition = */ cond,
215  /* Split Before = */ inst,
216  /* Unreachable = */ false);
217 
218  // Create a PHI at the split point.
219  builder.SetInsertPoint(inst);
220  auto *phi = builder.CreatePHI(inst->getType(), 2, "dee");
221 
222  // Move the instruction to the "then" basic block.
223  inst->moveBefore(then_terminator);
224 
225  // Replace uses of the insert result with the PHI node.
226  inst->replaceAllUsesWith(phi);
227 
228  // Patch up the PHI for the "then" branch.
229  auto *then_bb = then_terminator->getParent();
230  phi->addIncoming(inst, then_bb);
231 
232  // Patch up the PHI for the "else" branch.
233  auto &original_collection = insert_inst->getBaseCollection();
234  phi->addIncoming(&original_collection, original_bb);
235 
236  } else if (auto *swap_inst = into<SeqSwapInst>(inst)) {
237  warnln("Swap instruction is unimplemented!");
238  } else if (auto *swap_within_inst = into<SeqSwapWithinInst>(inst)) {
239  warnln("Swap instruction is unimplemented!");
240  }
241  }
242  }
243  }
244 
245  return true;
246  }
