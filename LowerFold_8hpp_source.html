40  static bool lower_fold(FoldInst &I) {
41 
42  // Split the instruction out of its basic block, such that:
43  auto *header = I.getParent();
44 
45  auto *preheader =
46  llvm::SplitBlock(header,
47  &I.getCallInst(),
48  /* DomTreeUpdater = */ (llvm::DomTreeUpdater *)nullptr,
49  /* LoopInfo = */ nullptr,
50  /* MemorySSAUpdater = */ nullptr,
51  /* BBName = */ "fold.loop.preheader.",
52  /* Before = */ true);
53 
54  auto *continuation =
55  llvm::SplitBlock(header,
56  I.getCallInst().getNextNode(),
57  /* DomTreeUpdater = */ (llvm::DomTreeUpdater *)nullptr,
58  /* LoopInfo = */ nullptr,
59  /* MemorySSAUpdater = */ nullptr,
60  /* BBName = */ "fold.continuation.",
61  /* Before = */ false);
62 
63  // Fetch information about the collection being folded over.
64  auto &collection = I.getCollection();
65  auto &collection_type =
66  MEMOIR_SANITIZE(dyn_cast_or_null<CollectionType>(type_of(collection)),
67  "FoldInst accessing non-collection type!");
68  bool collection_is_assoc = isa<AssocArrayType>(&collection_type);
69 
70  // Above the fold instruction, create a SizeInst for the collection,
71  MemOIRBuilder builder(I);
72 
73  auto *collection_size =
74  &builder.CreateSizeInst(&collection, "fold.size.")->getCallInst();
75 
76  // Create a compare instruction to check if the collection is empty.
77  auto *collection_nonempty = builder.CreateICmpEQ(collection_size,
78  builder.getInt64(0),
79  "fold.nonempty.");
80 
81  // Split the header after the fold, creating an if-else based on the
82  // comparison. The block will be inserted in the else side of the branch.
83  auto *else_block =
84  llvm::SplitBlockAndInsertIfElse(/* Cond = */ collection_nonempty,
85  /* SplitBefore = */ &I.getCallInst(),
86  /* Unreachable = */ false);
87 
88  // If the collection is associative, create an AssocKeysInst.
89  auto *iterable = &collection;
90  if (collection_is_assoc) {
91  // Make sure the builder's insertion point is correct.
92  builder.SetInsertPoint(&I.getCallInst());
93 
94  // Create the AssocKeysInst
95  auto &keys = MEMOIR_SANITIZE(builder.CreateAssocKeysInst(&collection),
96  "Failed to create AssocKeysInst!");
97 
98  // Set the iterable collection to the AssocKeysInst.
99  iterable = &keys.getCallInst();
100  }
101 
102  // Split the block before the fold instruction, creating a for-loop from 0
103  // to collection_size.
104  auto [loop_insertion_point, index] =
105  llvm::SplitBlockAndInsertSimpleForLoop(collection_size,
106  &I.getCallInst());
107 
108  // Update the builder's insertion point to the index PHI.
109  auto &index_phi =
110  MEMOIR_SANITIZE(dyn_cast_or_null<llvm::PHINode>(index),
111  "Index of for loop is not a PHI, LLVM Lied!");
112  builder.SetInsertPoint(&index_phi);
113 
114  // Create a PHI node for the accumulator value.
115  auto *accumulator_llvm_type = I.getCallInst().getType();
116  auto &accumulator = MEMOIR_SANITIZE(
117  builder.CreatePHI(accumulator_llvm_type, 2, "fold.accum.phi."),
118  "Unable to create PHI node for accumulator!");
119 
120  // Update the builder's insertion point to the start of the loop.
121  builder.SetInsertPoint(loop_insertion_point);
122 
123  // If the collection is associative, read the key for this iteration.
124  llvm::Value *key = index;
125  if (collection_is_assoc) {
126  // Fetch the key type.
127  auto *assoc_type = cast<AssocArrayType>(&collection_type);
128  auto &key_type = assoc_type->getKeyType();
129 
130  // Read the key from the keys sequence.
131  auto &read_key = MEMOIR_SANITIZE(
132  builder.CreateIndexReadInst(key_type, iterable, index),
133  "Failed to create IndexReadInst for AssocKeys!");
134 
135  // Save the key for later.
136  key = &read_key.getValueRead();
137  }
138 
139  // Insert a Get/ReadInst at the beginning of the loop.
140  llvm::Value *element;
141  auto &element_type = collection_type.getElementType();
142  if (isa<StructType>(&element_type)) {
143  if (collection_is_assoc) {
144  // Read the value from the collection.
145  auto &read_value = MEMOIR_SANITIZE(
146  builder.CreateAssocGetInst(element_type, &collection, key),
147  "Failed to create AssocGetInst!");
148 
149  // Save the element read for later.
150  element = &read_value.getCallInst();
151 
152  } else {
153  // Read the value from the collection.
154  auto &read_value = MEMOIR_SANITIZE(
155  builder.CreateIndexGetInst(element_type, &collection, key),
156  "Failed to create IndexGetInst!");
157 
158  // Save the element read for later.
159  element = &read_value.getCallInst();
160  }
161  } else {
162  if (collection_is_assoc) {
163  // Read the value from the collection.
164  auto &read_value = MEMOIR_SANITIZE(
165  builder.CreateAssocReadInst(element_type, &collection, key),
166  "Failed to create AssocReadInst!");
167 
168  // Save the element read for later.
169  element = &read_value.getCallInst();
170 
171  } else {
172  // Read the value from the collection.
173  auto &read_value = MEMOIR_SANITIZE(
174  builder.CreateIndexReadInst(element_type, &collection, key),
175  "Failed to create IndexReadInst!");
176 
177  // Save the element read for later.
178  element = &read_value.getCallInst();
179  }
180  }
181 
182  // Fetch information about the fold function
183  auto &function = I.getFunction();
184  auto *function_type = function.getFunctionType();
185 
186  // Construct the list of arguments to pass into the function.
187  vector<llvm::Value *> arguments = { &accumulator, key, element };
188  for (unsigned closed_idx = 0; closed_idx < I.getNumberOfClosed();
189  ++closed_idx) {
190 
191  // Get the closed variable.
192  auto &closed = I.getClosed(closed_idx);
193 
194  // Add the closed variable to the list of arguments.
195  arguments.push_back(&closed);
196  }
197 
198  // Create a call to the fold function so that we can inline it.
199  auto &call = MEMOIR_SANITIZE(
200  builder.CreateCall(function_type, &function, arguments, "fold.accum."),
201  "Failed to create call to the FoldInst function!");
202 
203  // Replace all uses of the fold instruction result with the call result.
204  I.getCallInst().replaceAllUsesWith(&call);
205 
206  // Split the call into its own basic block.
207  auto *loop_body = call.getParent();
208 
209  llvm::SplitBlock(loop_body,
210  &call,
211  /* DomTreeUpdater = */ (llvm::DomTreeUpdater *)nullptr,
212  /* LoopInfo = */ nullptr,
213  /* MemorySSAUpdater = */ nullptr,
214  /* BBName = */ "fold.loop.header.",
215  /* Before = */ true);
216 
217  llvm::SplitBlock(loop_body,
218  call.getNextNode(),
219  /* DomTreeUpdater = */ (llvm::DomTreeUpdater *)nullptr,
220  /* LoopInfo = */ nullptr,
221  /* MemorySSAUpdater = */ nullptr,
222  /* BBName = */ "fold.loop.check.",
223  /* Before = */ false);
224 
225  loop_body->setName("fold.loop.body.");
226 
227  // Find the loop pre-header by finding which of the incoming values for
228  // the index PHI is a constant zero.
229  llvm::BasicBlock *loop_preheader = nullptr;
230  auto *loop_header = index_phi.getParent();
231  for (auto *pred : llvm::predecessors(loop_header)) {
232  auto *index_incoming = index_phi.getIncomingValueForBlock(pred);
233 
234  // If the incoming index value is a constant, we found our pre-header.
235  if (isa<llvm::ConstantInt>(index_incoming)) {
236  loop_preheader = pred;
237  break;
238  }
239  }
240  MEMOIR_NULL_CHECK(loop_preheader,
241  "Unable to find preheader of the loop we created!");
242 
243  // For each predecessor of the loop header:
244  // - If it _is_ the the loop pre-header, set the incoming value of the
245  // accumulator to be the initial operand of the FoldInst.
246  // - If it is _not_ the loop pre-header, set the incoming value of the
247  // accumulator to be the result of the call.
248  for (auto *pred : llvm::predecessors(loop_header)) {
249  if (pred == loop_preheader) {
250  accumulator.addIncoming(&I.getInitial(), loop_preheader);
251  } else {
252  accumulator.addIncoming(&call, pred);
253  }
254  }
255 
256  // Now, inline the fold function.
257  llvm::InlineFunctionInfo IFI;
258  auto inline_result = llvm::InlineFunction(call, IFI);
259  if (not inline_result.isSuccess()) {
260  println("Inlining fold function failed!");
261  println(" Reason: ", inline_result.getFailureReason());
262  }
263 
264  return true;
265  }

267  bool transform() {
268  // Collect all fold instructions to be lowered.
269  vector<FoldInst *> folds = {};
270  for (auto &F : M) {
271  for (auto &BB : F) {
272  for (auto &I : BB) {
273  if (auto *fold = into<FoldInst>(I)) {
274  folds.push_back(fold);
275  }
276  }
277  }
278  }
279 
280  // If there are no folds, return false.
281  if (folds.size() == 0) {
282  return false;
283  }
284 
285  // Lower each fold.
286  for (auto *fold : folds) {
287  if (lower_fold(*fold)) {
288  this->to_cleanup.insert(&fold->getCallInst());
289  }
290  }
291 
292  // If we got this far, we modified the code. Return true.
293  return true;
294  }
295 
296  bool cleanup() {
297  if (this->to_cleanup.empty()) {
298  return false;
299  }
300 
301  // TODO: erase instructions from parent
302  for (auto *inst : this->to_cleanup) {
303  inst->eraseFromParent();
304  }
305 
306  return true;
307  }
308 
309 private:
310  // Owned state.
311  bool transformed;
312 
313  // Borrowed state.
314  llvm::Module &M;
315  set<llvm::Instruction *> to_cleanup;
316 };
