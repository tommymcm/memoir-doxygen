68  static inline bool value_escapes(llvm::Value &V) {
69  for (auto *user : V.users()) {
70  if (auto *user_as_call = dyn_cast_or_null<llvm::CallBase>(user)) {
71  // If the user is a memoir instruction, continue.
72  if (auto *user_as_memoir = MemOIRInst::get(*user_as_call)) {
73  continue;
74  }
75 
76  // Get the called function.
77  auto *called_function = user_as_call->getCalledFunction();
78 
79  // If it is an indirect call, then the value escapes.
80  // TODO: improve this by using NOELLE's complete call graph
81  if (called_function == nullptr) {
82  return true;
83  }
84 
85  // If it is empty, then the value escapes.
86  if (called_function->empty()) {
87  return true;
88  }
89  }
90  }
91 
92  return false;
93  }

95  static inline void analyze_value(llvm::Value &V,
96  StructToTypeMapTy &structs_of_type,
97  StructTypeSetTy &escaped_types) {
98  // Check if this value is a struct type.
99  if (!Type::value_is_struct_type(V)) {
100  return;
101  }
102 
103  infoln("Found struct: ", V);
104 
105  // Get the type of this struct.
106  auto *type = TypeAnalysis::analyze(V);
107  auto *struct_type = dyn_cast_or_null<StructType>(type);
108  MEMOIR_NULL_CHECK(struct_type,
109  "Could not determine the StructType of a struct value!");
110 
111  // Check if the struct escapes.
112  if (value_escapes(V)) {
113  infoln("Value escapes! ", V);
114  escaped_types.insert(struct_type);
115  return;
116  }
117 
118  // Otherwise, record this struct value.
119  structs_of_type[struct_type].insert(&V);
120 
121  return;
122  }

124  // Analysis
125  LiveFieldMapTy analyze(llvm::Module &M) {
126  LiveFieldMapTy live_fields = {};
127 
128  // Find all structs.
129  StructToTypeMapTy structs_of_type = {};
130  StructTypeSetTy escaped_types = {};
131 
132  // For each function:
133  for (auto &F : M) {
134  // Ignore function declarations.
135  if (F.empty()) {
136  continue;
137  }
138 
139  // Check the arguments.
140  for (auto &A : F.args()) {
141  analyze_value(A, structs_of_type, escaped_types);
142  }
143 
144  // Check the function.
145  for (auto &BB : F) {
146  for (auto &I : BB) {
147  analyze_value(I, structs_of_type, escaped_types);
148  }
149  }
150  }
151 
152  debugln("Got structs and their types.");
153 
154  // For each struct type:
155  for (auto const &[struct_type, struct_values] : structs_of_type) {
156  auto &index_to_access_map = live_fields[struct_type];
157  // Inspect the values of its type.
158  for (auto *value : struct_values) {
159  // Get all users of this values.
160  for (auto *user : value->users()) {
161  // Only need to get direct users, we've already gathered each named
162  // variable so PHIs will be handled on their own.
163  auto *user_as_inst = dyn_cast<llvm::Instruction>(user);
164  if (user_as_inst == nullptr) {
165  continue;
166  }
167 
168  // Only need to inspect memoir instructions.
169  auto *user_as_memoir = MemOIRInst::get(*user_as_inst);
170 
171  // Inspect struct accesses.
172  if (auto *user_as_access =
173  dyn_cast_or_null<AccessInst>(user_as_memoir)) {
174 
175  // Get the field index.
176  unsigned field_index;
177  if (auto *user_as_struct_read =
178  dyn_cast<StructReadInst>(user_as_access)) {
179  field_index = user_as_struct_read->getFieldIndex();
180  } else if (auto *user_as_struct_write =
181  dyn_cast_or_null<StructWriteInst>(user_as_access)) {
182  field_index = user_as_struct_write->getFieldIndex();
183  } else if (auto *user_as_struct_get =
184  dyn_cast_or_null<StructGetInst>(user_as_access)) {
185  field_index = user_as_struct_get->getFieldIndex();
186  }
187 
188  // Register the field index as being alive with this access.
189  index_to_access_map[field_index].insert(user_as_access);
190  }
191 
192  // If it's not a struct access, we have nothing to look at.
193  }
194  }
195  }
196 
197  debugln("Done analyzing");
198 
199  return live_fields;
200  }

202  // Transformation
203  bool transform(LiveFieldMapTy &&live_fields) {
204  bool transformed = false;
205 
206  set<llvm::Instruction *> instructions_to_delete = {};
207 
208  // For each struct type:
209  for (auto const &[struct_type, field_index_to_accesses] : live_fields) {
210  debugln("Transforming ", *struct_type);
211 
212  // Get the type definition as an instruction.
213  auto &type_definition = struct_type->getDefinition();
214  set<unsigned> fields_to_remove = {};
215 
216  // For each field in the struct definition:
217  unsigned num_fields = struct_type->getNumFields();
218  debugln(" # fields = ", num_fields);
219  vector<llvm::Value *> fields_after_removal; // this is in reverse order.
220  fields_after_removal.reserve(num_fields + 2);
221  for (int16_t field_idx = num_fields - 1; field_idx >= 0; field_idx--) {
222  // debugln(" inspecting index ", field_idx);
223 
224  // If the field is alive, we have nothing to do.
225  auto found_field = field_index_to_accesses.find(field_idx);
226  if (found_field != field_index_to_accesses.end()) {
227  fields_after_removal.push_back(
228  &type_definition.getFieldTypeOperand(field_idx));
229  continue;
230  }
231 
232  transformed |= true;
233 
234  // Otherwise, we need to:
235  // 1. mark this field for removal.
236  fields_to_remove.insert(field_idx);
237 
238  // 2. update the field index of all accesses.
239  for (unsigned orig_field_idx = field_idx + 1;
240  orig_field_idx < num_fields;
241  orig_field_idx++) {
242  debugln(" updating index ", orig_field_idx);
243 
244  auto found_orig_field = field_index_to_accesses.find(orig_field_idx);
245  if (found_orig_field == field_index_to_accesses.end()) {
246  continue;
247  }
248 
249  for (auto *access : found_orig_field->second) {
250  unsigned access_field_index;
251  llvm::Use *access_field_index_use;
252  if (auto *struct_read = dyn_cast<StructReadInst>(access)) {
253  access_field_index = struct_read->getFieldIndex();
254  access_field_index_use =
255  &struct_read->getFieldIndexOperandAsUse();
256  } else if (auto *struct_write = dyn_cast<StructWriteInst>(access)) {
257  access_field_index = struct_write->getFieldIndex();
258  access_field_index_use =
259  &struct_write->getFieldIndexOperandAsUse();
260  } else if (auto *struct_get = dyn_cast<StructGetInst>(access)) {
261  access_field_index = struct_get->getFieldIndex();
262  access_field_index_use = &struct_get->getFieldIndexOperandAsUse();
263  } else {
264  MEMOIR_UNREACHABLE(
265  "We have a field access that is no longer a Struct{Read,Write,Get}Inst!");
266  }
267 
268  // If the field index is less than that of the field being
269  // deleted, do nothing.
270  if (access_field_index < field_idx) {
271  continue;
272  }
273 
274  // Get the constant for field index-1.
275  auto *field_index_type = access_field_index_use->get()->getType();
276  auto *new_field_index_value =
277  llvm::ConstantInt::get(field_index_type,
278  access_field_index - 1);
279 
280  // Replace the field index in the access.
281  access_field_index_use->set(new_field_index_value);
282  }
283  }
284  }
285 
286  // If there are fields to remove, let's update the struct type
287  // definition.
288  if (fields_to_remove.size() > 0) {
289  // Get information about the type definition.
290  auto &type_definition_as_call = type_definition.getCallInst();
291 
292  // Get the new num fields constant integer.
293  auto &num_fields_as_value = type_definition.getNumberOfFieldsOperand();
294  auto *field_index_type = num_fields_as_value.getType();
295 
296  auto new_num_fields = num_fields - fields_to_remove.size();
297  auto *new_num_fields_value =
298  llvm::ConstantInt::get(field_index_type, new_num_fields);
299  fields_after_removal.push_back(new_num_fields_value);
300 
301  // Get the name of the struct type.
302  auto &type_name = type_definition.getNameOperand();
303  fields_after_removal.push_back(&type_name);
304 
305  // Construct a new call to replace this one.
306  auto *function_type = type_definition_as_call.getFunctionType();
307  auto *function = type_definition_as_call.getCalledFunction();
308  auto *new_type_definition = CallInst::Create(
309  function_type,
310  function,
311  vector<llvm::Value *>(fields_after_removal.rbegin(),
312  fields_after_removal.rend()),
313  /*NameStr=*/"dfe.",
314  /*InsertBefore=*/&type_definition_as_call);
315 
316  // Replace the old call with this one.
317  type_definition_as_call.replaceAllUsesWith(new_type_definition);
318 
319  // Mark old one for cleanup.
320  instructions_to_delete.insert(&type_definition_as_call);
321  }
322  }
323 
324  // Cleanup instructions.
325  for (auto *inst : instructions_to_delete) {
326  inst->eraseFromParent();
327  }
328 
329  return transformed;
330  }
