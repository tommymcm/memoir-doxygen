88  static inline bool value_escapes(llvm::Value &V) {
89  for (auto *user : V.users()) {
90  if (auto *user_as_call = dyn_cast_or_null<llvm::CallBase>(user)) {
91  // If the user is a memoir instruction, continue.
92  if (auto *user_as_memoir = MemOIRInst::get(*user_as_call)) {
93  continue;
94  }
95 
96  // Get the called function.
97  auto *called_function = user_as_call->getCalledFunction();
98 
99  // If it is an indirect call, then the value escapes.
100  // TODO: improve this by using NOELLE's complete call graph
101  if (called_function == nullptr) {
102  return true;
103  }
104 
105  // If it is empty, then the value escapes.
106  if (called_function->empty()) {
107  return true;
108  }
109  }
110  }
111 
112  return false;
113  }

115  static inline void analyze_value(llvm::Value &V,
116  FieldsToElideMapTy &fields_to_elide,
117  TypeToStructsMapTy &structs_of_type,
118  StructTypeSetTy &escaped_types) {
119  // Check if this value is a struct type.
120  if (!Type::value_is_struct_type(V)) {
121  return;
122  }
123 
124  infoln("Found struct: ", V);
125 
126  // Get the type of this struct.
127  auto *type = TypeAnalysis::analyze(V);
128  auto *struct_type = dyn_cast_or_null<StructType>(type);
129  MEMOIR_NULL_CHECK(struct_type,
130  "Could not determine the StructType of a struct value!");
131 
132 // TODO: remove this after done with testing!
133 #if FIELD_ELISION_TEST
134  auto &struct_fields_to_elide = fields_to_elide[struct_type];
135  bool found = false;
136  for (auto candidate : struct_fields_to_elide) {
137  if (candidate.find(0) != candidate.end()) {
138  found = true;
139  break;
140  }
141  }
142  if (!found) {
143  fields_to_elide[struct_type].push_back({ 0 });
144  }
145 #endif
146 
147  // Check if this struct is of a type we care about.
148  if (fields_to_elide.find(struct_type) == fields_to_elide.end()) {
149  return;
150  }
151 
152  // Check if the struct escapes.
153  if (value_escapes(V)) {
154  infoln("Value escapes! ", V);
155  escaped_types.insert(struct_type);
156  return;
157  }
158 
159  // Otherwise, record this struct value.
160  structs_of_type[struct_type].insert(&V);
161 
162  return;
163  }

165  // Analysis
166  FieldsToAccessesMapTy analyze(llvm::Module &M,
167  FieldsToElideMapTy &fields_to_elide) {
168  FieldsToAccessesMapTy fields_to_accesses = {};
169 
170  // If there are no fields to elide, return.
171  if (fields_to_elide.empty()) {
172  return fields_to_accesses;
173  }
174 
175  // Find all structs.
176  map<llvm::Function *, TypeToStructsMapTy> structs_of_type = {};
177  StructTypeSetTy escaped_types = {};
178 
179  // For each function:
180  for (auto &F : M) {
181  // Ignore function declarations.
182  if (F.empty()) {
183  continue;
184  }
185 
186  auto &structs_of_type_in_function = structs_of_type[&F];
187 
188  // Gather struct arguments for the types we care about.
189  for (auto &A : F.args()) {
190  analyze_value(A,
191  fields_to_elide,
192  structs_of_type_in_function,
193  escaped_types);
194  }
195 
196  // Gather struct variables for the types we care about.
197  for (auto &BB : F) {
198  for (auto &I : BB) {
199  analyze_value(I,
200  fields_to_elide,
201  structs_of_type_in_function,
202  escaped_types);
203  }
204  }
205  }
206 
207  infoln("Got structs and their types.");
208 
209  // Filter out the struct types that escaped.
210  for (auto *escaped_type : escaped_types) {
211  auto found = fields_to_elide.find(escaped_type);
212  if (found != fields_to_elide.end()) {
213  fields_to_elide.erase(found);
214  }
215  }
216 
217  // Filter out the struct types that had no variables.
218  // for (auto it = fields_to_elide.begin(); it != fields_to_elide.end();
219  // ++it)
220  // {
221  // if (structs_of_type.find(it->first) == structs_of_type.end()) {
222  // it = fields_to_elide.erase(it);
223  // }
224  // }
225 
226  infoln("Filtered escaped and unused struct types.");
227 
228  // For each function:
229  for (auto &F : M) {
230  auto &fields_to_accesses_for_function = fields_to_accesses[&F];
231 
232  // For each struct type that has a field elision candidate:
233  for (auto const &[struct_type, elision_candidates] : fields_to_elide) {
234  // Get the struct variables of this type.
235  auto &struct_variables = structs_of_type[&F][struct_type];
236 
237  // Get the access map for this type.
238  auto &index_to_access_map =
239  fields_to_accesses_for_function[struct_type];
240 
241  // For each elision grouping:
242  for (auto candidate : elision_candidates) {
243  // Collect all accesses to update.
244  for (auto *struct_variable : struct_variables) {
245  // Get all users of this values.
246  for (auto *user : struct_variable->users()) {
247  // Only need to get direct users, we've already gathered each
248  // named variable so PHIs will be handled on their own.
249  auto *user_as_inst = dyn_cast<llvm::Instruction>(user);
250  if (user_as_inst == nullptr) {
251  continue;
252  }
253 
254  // Only need to inspect memoir instructions.
255  auto *user_as_memoir = MemOIRInst::get(*user_as_inst);
256 
257  // Inspect struct accesses.
258  if (auto *user_as_access =
259  dyn_cast_or_null<AccessInst>(user_as_memoir)) {
260 
261  // Get the field index.
262  unsigned field_index;
263  if (auto *user_as_struct_read =
264  dyn_cast<StructReadInst>(user_as_access)) {
265  field_index = user_as_struct_read->getFieldIndex();
266  } else if (auto *user_as_struct_write =
267  dyn_cast_or_null<StructWriteInst>(
268  user_as_access)) {
269  field_index = user_as_struct_write->getFieldIndex();
270  } else if (auto *user_as_struct_get =
271  dyn_cast_or_null<StructGetInst>(
272  user_as_access)) {
273  field_index = user_as_struct_get->getFieldIndex();
274  }
275 
276  // Register the field index as being alive with this access.
277  index_to_access_map[field_index].insert(user_as_access);
278  }
279 
280  // If it's not a struct access, we have nothing to look at.
281  }
282  }
283  }
284  }
285  }
286 
287  infoln("Done analyzing");
288 
289  return fields_to_accesses;
290  }

325  using UsedFieldsSetTy = multimap<StructType *, unsigned>;
326  UsedFieldsSetTy &traverse_call_graph_node(
327  map<llvm::Function *, UsedFieldsSetTy> &function_to_used_fields,
328  FieldsToAccessesMapTy &fields_to_accesses,
329  set<llvm::CallGraphNode *> visited,
330  llvm::CallGraphNode &node) {
331  // Mark this node as visited.
332  visited.insert(&node);
333 
334  // Get node's information.
335  auto *node_function = node.getFunction();
336 
337  // Get the set of used fields.
338  auto &used_fields = function_to_used_fields[node_function];
339 
340  // Check for corner cases.
341  if (node_function == nullptr) {
342  // warnln("Node function is NULL!");
343  return used_fields;
344  } else if (FunctionNames::is_memoir_call(*node_function)) {
345  return used_fields;
346  } else if (node_function->empty()) {
347  // warnln("Node function is defined externally");
348  return used_fields;
349  }
350 
351  debugln("CallGraph: Visiting ", node_function->getName());
352 
353  for (auto const &[function, callee_node] : node) {
354  // Get the callee function.
355  auto *callee_function = callee_node->getFunction();
356 
357  // If the callee is NULL, just ignore it. We have already checked for
358  // structs escaping along indirect call edges. This may be naive, but
359  // until we find a case that we need to handle it will remain that way.
360  if (callee_function == nullptr) {
361  warnln("Callee function is NULL within ", node_function->getName());
362  continue;
363  };
364 
365  // Check if this node has already visited.
366  if (visited.find(callee_node) != visited.end()) {
367  // Insert the elements from this node into our own used types.
368  auto &callee_used_fields = function_to_used_fields[callee_function];
369  for (auto const &[struct_type, field_index] : callee_used_fields) {
370  insert_unique(used_fields, struct_type, field_index);
371  }
372  continue;
373  }
374 
375  // If we haven't visited it, then visit the callee.
376  auto &callee_used_fields =
377  traverse_call_graph_node(function_to_used_fields,
378  fields_to_accesses,
379  visited,
380  *callee_node);
381  for (auto const &[struct_type, field_index] : callee_used_fields) {
382  insert_unique(used_fields, struct_type, field_index);
383  }
384  }
385 
386  // Insert all fields used by this function.
387  for (auto const &[struct_type, field_to_accesses] :
388  fields_to_accesses[node_function]) {
389  for (auto const &[field_index, accesses] : field_to_accesses) {
390  if (accesses.size() > 0) {
391  insert_unique(used_fields, struct_type, field_index);
392  }
393  }
394  }
395 
396  return used_fields;
397  }

400  llvm::CallGraph &CG,
401  FieldsToAccessesMapTy &fields_to_accesses,
402  const llvm::Function &entry_function) {
403 
404  map<llvm::Function *, UsedFieldsSetTy> function_to_used_fields = {};
405 
406  auto &entry_node = MEMOIR_SANITIZE(
407  CG[&entry_function],
408  "Could not get the CallGraphNode for the entry function.");
409 
410  set<llvm::CallGraphNode *> visited = {};
411  traverse_call_graph_node(function_to_used_fields,
412  fields_to_accesses,
413  visited,
414  entry_node);
415 
416  return function_to_used_fields;
417  }

419  bool transform(llvm::Module &M,
420  llvm::CallGraph &CG,
421  FieldsToElideMapTy &fields_to_elide,
422  FieldsToAccessesMapTy fields_to_accesses) {
423  bool transformed = false;
424 
425  // If there are no fields to elide, return.
426  if (fields_to_elide.empty()) {
427  return transformed;
428  }
429 
430  set<llvm::Instruction *> instructions_to_delete = {};
431 
432  llvm::Type *llvm_collection_type = nullptr;
433 
434  // Now for the tricky bit.
435  // We need to version all functions in the callgraph so that there is a
436  // route from the entry function to each function where an access resides.
437  auto &entry_function = MEMOIR_SANITIZE(
438  M.getFunction("main"),
439  "Could not get the main function, FieldElision assumes whole-program.");
440 
441  // Perform a post-order DFS traversal of the callgraph, collecting the
442  // types that must be propagated along
443  map<llvm::Function *, UsedFieldsSetTy> function_to_used_fields =
444  traverse_call_graph(CG, fields_to_accesses, entry_function);
445 
446  // Collect the elided fields that will be added to each function's
447  // list of parameters.
448  // Map from function to a (struct_type, candidates[i])
449  map<llvm::Function *, ordered_multimap<StructType *, size_t>>
450  function_to_argument_order = {};
451  for (auto const &[function, used_fields] : function_to_used_fields) {
452  // Skip this if there are no used fields.
453  if (used_fields.empty()) {
454  continue;
455  }
456 
457  debugln("Function ", function->getName(), " uses the following fields");
458  auto &argument_order = function_to_argument_order[function];
459 
460  // Check for any fields that are elision candidates.
461  for (auto const &[struct_type, candidates] : fields_to_elide) {
462  auto used_fields_for_type = used_fields.equal_range(struct_type);
463 
464  for (auto it = used_fields_for_type.first;
465  it != used_fields_for_type.second;
466  ++it) {
467 
468  auto field_index = it->second;
469 
470  debugln(" Field ", field_index, " of ", struct_type->getName());
471 
472  size_t candidate_index = 0;
473  for (auto candidate : candidates) {
474  for (auto candidate_field_index : candidate) {
475  if (field_index == candidate_field_index) {
476  debugln(" is a candidate (",
477  candidate_index,
478  ") for elision");
479  insert_unique(argument_order, struct_type, candidate_index);
480  break;
481  }
482  }
483  ++candidate_index;
484  }
485  }
486  }
487  }
488 
489  // For each function that has a used type, we need to change its function
490  // type to include the replaced assoc as arguments.
491  map<llvm::Function *, map<StructType *, map<size_t, llvm::Value *>>>
492  function_to_elision_values = {};
493  map<llvm::Function *, llvm::Function *> function_clones = {};
494  for (auto const &[function, elision_parameters] :
495  function_to_argument_order) {
496  // Skip main, it is the entry function.
497  if (function->getName() == "main") {
498  continue;
499  }
500 
501  if (function == nullptr || function->empty()) {
502  continue;
503  }
504 
505  // Get information from the old function type.
506  auto &old_function_type =
507  MEMOIR_SANITIZE(function->getFunctionType(),
508  "Couldn't determine the function type");
509  auto *return_type = old_function_type.getReturnType();
510  auto param_types = old_function_type.params().vec();
511  auto is_var_arg = old_function_type.isVarArg();
512 
513  // Get the memoir Collection* type.
514  auto *collection_ptr_type =
515  FunctionNames::get_memoir_function(M, MemOIR_Func::SIZE)
516  ->getFunctionType()
517  ->getParamType(0);
518  MEMOIR_NULL_CHECK(collection_ptr_type, "Collection* type is NULL!");
519 
520  // Add the new argument type.
521  auto new_arg_index = param_types.size();
522  for (auto elision_param : elision_parameters) {
523  param_types.push_back(collection_ptr_type);
524  }
525 
526  // Construct the new function type.
527  auto *new_function_type =
528  FunctionType::get(return_type, param_types, is_var_arg);
529  MEMOIR_NULL_CHECK(new_function_type,
530  "Could not get the new function type");
531 
532  // Create an empty function to clone into.
533  auto *new_function = Function::Create(new_function_type,
534  function->getLinkage(),
535  function->getName(),
536  M);
537  MEMOIR_NULL_CHECK(new_function, "Could not create new function!");
538 
539  debugln("Cloning ", function->getName());
540 
541  // Version the function.
542  ValueToValueMapTy vmap;
543  for (auto &old_argument : function->args()) {
544  auto *new_argument =
545  new_function->arg_begin() + old_argument.getArgNo();
546  vmap.insert(std::make_pair(&old_argument, new_argument));
547  }
548 
549  llvm::SmallVector<llvm::ReturnInst *, 8> returns;
550  llvm::CloneFunctionInto(new_function, function, vmap, false, returns);
551 
552  // Collect the elision arguments.
553  auto arg_idx = new_arg_index;
554  for (auto const &[elision_type, elision_group] : elision_parameters) {
555  function_to_elision_values[new_function][elision_type][elision_group] =
556  (new_function->arg_begin() + arg_idx++);
557  }
558 
559  // Record the clone.
560  function_clones[function] = new_function;
561  }
562 
563  // Allocate the associative arrays for each elision candidate in the main
564  // function.
565  auto &entry_bb = entry_function.getEntryBlock();
566  auto *entry_insertion_point = entry_bb.getFirstNonPHI();
567  MemOIRBuilder builder(entry_insertion_point);
568  for (auto const &[struct_type, candidates] : fields_to_elide) {
569  auto &struct_ref_type = Type::get_ref_type(*struct_type);
570  auto *struct_ref_type_inst = builder.CreateTypeInst(struct_ref_type);
571  auto candidate_idx = 0;
572  for (auto candidate : candidates) {
573  if (candidate.size() == 1) {
574  // Create the Value type.
575  auto candidate_field_index = *(candidate.begin());
576  auto &candidate_field_type =
577  struct_type->getFieldType(candidate_field_index);
578  auto *candidate_type_inst =
579  builder.CreateTypeInst(candidate_field_type);
580 
581  // Allocate the associative arrays.
582  auto *allocation =
583  builder.CreateAssocArrayAllocInst(struct_ref_type,
584  candidate_field_type);
585 
586  function_to_elision_values[&entry_function][struct_type]
587  [candidate_idx] =
588  &allocation->getCallInst();
589  } else {
590  MEMOIR_UNREACHABLE("Field elision groups are not supported!");
591  }
592 
593  ++candidate_idx;
594  }
595  }
596 
597  // Iterate over each function, replacing calls with their clones and adding
598  // the relevant arguments.
599  for (auto &F : M) {
600  // Skip functions that have been cloned.
601  if (function_clones.find(&F) != function_clones.end()) {
602  continue;
603  }
604 
605  // Get the elision arguments for this function.
606  auto &argument_order = function_to_argument_order[&F];
607  auto &elision_values = function_to_elision_values[&F];
608 
609  // TODO
610  // Insert type annotations for the arguments.
611 
612  // Collect the calls and accesses.
613  set<llvm::CallBase *> calls = {};
614  set<AccessInst *> accesses = {};
615  for (auto &BB : F) {
616  for (auto &I : BB) {
617  if (auto *memoir_inst = MemOIRInst::get(I)) {
618  auto *access_inst = dyn_cast<AccessInst>(memoir_inst);
619  if (access_inst == nullptr) {
620  continue;
621  }
622 
623  if (isa<StructReadInst>(access_inst)
624  || isa<StructWriteInst>(access_inst)
625  || isa<StructGetInst>(access_inst)) {
626  accesses.insert(access_inst);
627  }
628  } else if (auto *call = dyn_cast<llvm::CallBase>(&I)) {
629  calls.insert(call);
630  }
631  }
632  }
633 
634  // Update the calls.
635  for (auto *call : calls) {
636  // Get the called function.
637  auto *called_function = call->getCalledFunction();
638 
639  // Skip external functions.
640  if (called_function == nullptr) {
641  continue;
642  }
643 
644  // Get the clone of the called function.
645  auto found_clone = function_clones.find(called_function);
646  if (found_clone == function_clones.end()) {
647  // If the called function was not cloned, skip it.
648  continue;
649  }
650  auto *cloned_function = found_clone->second;
651 
652  debugln(" clone= ", cloned_function->getName());
653 
654  // Get the order of arguments that this function needs.
655  auto &called_argument_order =
656  function_to_argument_order[called_function];
657 
658  // Update the list of arguments.
659  // TODO: fix this so it works with var args, this will currently be
660  // unaligned!
661  MEMOIR_ASSERT(!cloned_function->getFunctionType()->isVarArg(),
662  "Cloning function with var args is unsupported!");
663  vector<llvm::Value *> new_arguments(call->arg_begin(), call->arg_end());
664  for (auto const &[struct_type, elision_group] : called_argument_order) {
665  auto *elision_value = elision_values[struct_type][elision_group];
666  new_arguments.push_back(elision_value);
667  println("new arg: ", *elision_value);
668  }
669 
670  // Rebuild the call.
671  llvm::CallBase *new_call = nullptr;
672  if (auto *invoke_inst = dyn_cast<llvm::InvokeInst>(call)) {
673  new_call =

675  invoke_inst->getNormalDest(),

707  for (auto *access : accesses) {

709  unsigned field_index;
710  llvm::Use *field_index_as_use;
711  if (auto *struct_read = dyn_cast<StructReadInst>(access)) {
712  access_as_call = &struct_read->getCallInst();
713  field_index = struct_read->getFieldIndex();
714  field_index_as_use = &struct_read->getFieldIndexOperandAsUse();
715  } else if (auto *struct_write = dyn_cast<StructWriteInst>(access)) {
716  access_as_call = &struct_write->getCallInst();
717  field_index = struct_write->getFieldIndex();
718  field_index_as_use = &struct_write->getFieldIndexOperandAsUse();
719  } else if (auto *struct_get = dyn_cast<StructWriteInst>(access)) {
720  access_as_call = &struct_get->getCallInst();
721  field_index = struct_get->getFieldIndex();
722  field_index_as_use = &struct_get->getFieldIndexOperandAsUse();
723  } else {
724  continue;
725  }
726 
727  // Get the struct accessed.
728  auto &object_use = access->getObjectOperandAsUse();
729  auto *object_value = object_use.get();
730 
731  // Get the struct type.
732  auto *accessed_type = TypeAnalysis::analyze(*object_value);
733  auto *accessed_struct_type = cast<StructType>(accessed_type);
734 
735  // Check if this field index is elided.
736  size_t elision_group_index = (size_t)-1;
737  bool elided = false;
738  for (auto const &[struct_type, elision_groups] : fields_to_elide) {
739  if (struct_type == accessed_struct_type) {
740  elision_group_index = 0;
741  for (auto elision_group : elision_groups) {
742  for (auto elision_field_index : elision_group) {
743  if (elision_field_index == field_index) {
744  elided = true;
745  break;
746  }
747  }
748  if (elided) {
749  break;
750  }
751  ++elision_group_index;
752  }
753  if (elided) {
754  break;
755  }
756  }
757  }
758  if (!elided) {
759  continue;
760  }
761 
762  // Get the replacement function.
763  auto &original_function =
764  MEMOIR_SANITIZE(access_as_call->getCalledFunction(),
765  "Could not find struct function");
766  auto &replacement_function =
767  MEMOIR_SANITIZE(get_assoc_function(original_function),
768  "Could not find assoc function");
769 
770  // Get the elision replacement value.
771  auto *replacement_collection =
772  elision_values[accessed_struct_type][elision_group_index];
773 
774  // Replace the called function.
775  access_as_call->setCalledFunction(&replacement_function);
776 
777  // Replace the index with the struct being accessed.
778  field_index_as_use->set(object_use.get());
779 
780  // Replace the struct to the elided collection.
781  object_use.set(replacement_collection);
782  }
783  }
784

786  for (auto *inst : instructions_to_delete) {
787  inst->eraseFromParent();
788  }
789 
790  // Cleanup the originals.
791  for (auto [old_function, new_function] : function_clones) {
792  new_function->takeName(old_function);
793  old_function->eraseFromParent();
794  }
